Este índice no se regenera automáticamente. Para refrescarlo tras un batch:

```bash
# Conteo actualizado de pages
for d in concepts authors entities/works synthesis; do
  echo "$d: $(find wiki/$d -name '*.md' -not -name README.md | wc -l)"
done

# Cobertura corpus vs procesados

  -not -name "aggregation_stats.json" -not -name "blocks_filtered.json" \
  -not -name "discard_log.json" -not -name "pending_updates.json" \
  -not -name "promote_queue.json" -not -name "recommended_references.json" \
  -not -name "thesis_candidates.json" \
  | xargs -n1 basename | sed 's/\.json$//' | sort -u | wc -l
```
